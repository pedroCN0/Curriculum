create table expe(

cargo char(50),

CREATE TABLE IF NOT EXISTS `login` (
  `usuario` varchar(50) NOT NULL,
  `senha` varchar(20) NOT NULL,
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `loginusuario` char(11) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `loginusuario` (`loginusuario`)
);
